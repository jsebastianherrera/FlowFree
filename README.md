<h2>How to play ?</h2>
<table>
  <tr>
    <th>Key/keys</th>
    <th>Summary</th>
  </tr>
  <tr>
    <td>Arrow left</td>
    <td>Move to the left</td>
  </tr>
  <tr>
    <td>Arrow right</td>
    <td>Move to the right</td>
  </tr>
  <tr>
    <td>Arrow up</td>
    <td>Move one row up</td>
  </tr>
  <tr>
    <td>Arrow down</td>
    <td>Move one row down</td>
  </tr>
  <tr>
    <td>Enter </td>
    <td>Select current colour(It only works with main points)</td>
  </tr>
  <tr>
    <td>Ctrl + c </td>
    <td>Send signal to go back to the main menu</td>
  </tr>

</table>
